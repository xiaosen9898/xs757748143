// src/store/index.js
import Vue from 'vue';
import Vuex from 'vuex';

Vue.use(Vuex);

export default new Vuex.Store({
  state: {
    count: 0
  },
  mutations: {
    increment(state) {
      state.count++;
    },
    decrement(state) {
      state.count--;
    }
  },
  actions: {
    increment({ commit }) {
      commit('increment');
    },
    decrement({ commit }) {
      commit('decrement');
    }
  },
  getters: {
    doubleCount(state) {
      return state.count * 2;
    }
  }
});

TIP

  1. 定义：理解为数据
  2. 特点：
  1. 数据刷新会丢失；
  2. 一般结合localStorage来使用

  1. 注意：修改state的唯一方式是提交一个mutation，不能直接修改state

  2. 使用：

// src/store/index.js
import Vue from 'vue';
import Vuex from 'vuex';

Vue.use(Vuex);

export default new Vuex.Store({
  state: {
    count: 0,
    user: {
      name: 'Alice',
      age: 25
    }
  },
  getters: {                                  
    userName: (state) => state.user.name,     
    userAge: (state) => state.user.age,       
    doubleCount: (state) => state.count * 2
  },                                          
  mutations: {
    increment(state) {
      state.count++;
    }
  }
});

核心-actions ​

TIP

  1. 定义：action处理的是异步修改状态
  2. 特点：
  • mutation修改数据是同步的，action处理的是异步修改状态
  • 通过dispatch触发action
  1. 使用：
  1. store不分模块：
js
// src/store/index.js
import Vue from 'vue';
import Vuex from 'vuex';

Vue.use(Vuex);

export default new Vuex.Store({
  state: {
    name: 'Alice',
    user: null
  },
  mutations: {
    setName(state, newName) {
      state.name = newName;
    },
    setUser(state, user) {
      state.user = user;
    }
  },
  actions: {
    updateName({ commit }, newName) {
      setTimeout(() => {
        commit('setName', newName);
      }, 1000);
    },
    async fetchUser({ commit }) {
      const response = await fetch('https://api.example.com/user');
      const user = await response.json();
      commit('setUser', user);
    }
  }
});

// src/store/modules/user.js
const user = {
  namespaced: true,  // 使用命名空间
  state: {
    isLoggedIn: false,
    username: ''
  },
  mutations: {
    setLoginStatus(state, status) {
      state.isLoggedIn = status;
    },
    setUsername(state, username) {
      state.username = username;
    }
  },
  actions: {
    login({ commit }, username) {
      commit('setLoginStatus', true);
      commit('setUsername', username);
    },
    logout({ commit }) {
      commit('setLoginStatus', false);
      commit('setUsername', '');
    }
  }
};

export default user;

模块 modules ​

TIP

  1. 定义：
  1. 仓库数据多了，就用模块来划分；
  2. 每个模块可以有自己的state、getters、mutations、actions；
  3. 刷新的时候会重新创建一个仓库
  1. 注意：
  1. state默认是在模块内部的，其他属性是默认放在全局仓库的；
  2. 使用命名空间namespaced: true，就能把其他属性限制在模块中
