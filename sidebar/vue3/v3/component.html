import { ref } from 'vue';

    default: 'Hello'
  }
});

// 定义响应式的数据
const message = ref('子组件');

// 定义点击事件的处理函数

  message.value = '按钮已点击！';
};

2. 子传父 ​

  1. 子组件使用defineEmit()获取emit对象，通过emit触发事件，并且传递数据；
  2. 父组件监听子组件的事件，接收参数
vue
<template>
    <h2>子组件</h2>
    <!-- 按钮点击时触发 custom-event 事件 -->
    <button @click="changeCount">向父组件传递数据</button> 
</template>

<script setup>
// 通过 emit 触发事件，将数据传递给父组件
const emit = defineEmits(['change-count'])                

// 子组件的方法：触发事件并传递数据
const changeCount = () => {                               
  emit('change-count', 1000);                             
};                                                        
</script>

4. v-model ​

  1. 场景：父 ⇆ 子
  2. 本质：
  1. 组件标签上的v-model的本质：:moldeValue ＋ update:modelValue事件
  2. 如果v-model后面不使用自定变量 v-model，那么本质上父传的就是 :modelValue，监听的事件为@update:modelValue=""
  3. 如果v-model后面自定了变量 v-model:abc，那么本质上父传的就是 :abc，监听的事件为@update:abc=""
vue
<!-- ====== 父组件 ====== -->
<!-- 组件标签上使用v-model指令 -->
<MyInput v-model="userName"/>

<!-- 组件标签上v-model的本质 -->
<MyInput  :modelValue="userName" @update:model-value="userName = $event"/>


<!-- ====== 子组件 ====== -->
<template>
  <div class="box">
    <!--将接收的value值赋给input元素的value属性，目的是：为了呈现数据 -->
		<!--给input元素绑定原生input事件，触发input事件时，进而触发update:model-value事件-->
    <input 
       type="text" 
       :value="modelValue" 
       @input="emit('update:model-value',$event.target.value)"
    >
  </div>
</template>

<script setup lang="ts">
  // 接收props
  defineProps(['modelValue'])
  // 声明事件
  const emit = defineEmits(['update:model-value'])
</script>

5. $attrs ​

  1. 场景：爷 → 孙
  2. 说明：$attrs是一个对象，包含所有父组件传入的标签属性
  3. 注意：$attrs会自动排除props中声明的属性（可以认为声明过的 props 被子组件自己“消费”了）
vue
<!-- ====== 爷组件 ====== -->
<template>
  <div class="father">
    <h3>父组件</h3>
		<Child :a="a" :b="b" :c="c" :d="d" v-bind="{x:100,y:200}" :updateA="updateA"/>
  </div>
</template>

<script setup lang="ts" name="Father">
	import Child from './Child.vue'
	import { ref } from "vue";
	let a = ref(1)
	let b = ref(2)
	let c = ref(3)
	let d = ref(4)

	function updateA(value){
		a.value = value
	}
</script>

6. $refs、$parent ​

  1. 场景：$refs：父 → 子 $parent：子 → 父
  2. 原理：$attrs是一个对象，包含所有父组件传入的标签属性
  1. $refs：值为对象，包含所有被ref属性标识的DOM元素或组件实例
  2. $parent：值为对象，当前组件的父组件实例对象
  1. 注意：提供数据的一方需要向外暴露数据（defineExpose）

 

7. provide、inject ​

  1. 场景：爷 → 孙
  2. 使用：
  1. 在祖先组件中通过provide配置向后代组件提供数据
  2. 在后代组件中通过inject配置来声明接收数据
  1. 注意：传ref声明的数据时，应传响应式数据，也就是说不应该传.value
vue
<!-- ====== 爷组件 ====== -->
<template>
  <div class="father">
    <h3>父组件</h3>
    <h4>资产：{{ money }}</h4>
    <h4>汽车：{{ car }}</h4>
    <button @click="money += 1">资产+1</button>
    <button @click="car.price += 1">汽车价格+1</button>
    <Child/>
  </div>
</template>

<script setup lang="ts" name="Father">
  import Child from './Child.vue'
  import { ref,reactive,provide } from "vue";
  // 数据
  let money = ref(100)
  let car = reactive({
    brand:'奔驰',
    price:100
  })
  // 用于更新money的方法
  function updateMoney(value:number){
    money.value += value
  }
  // 提供数据
  provide('moneyContext',{money,updateMoney})
  provide('car',car)
</script>

2. 具名插槽 ​

  1. 需求：一个组件内有多处，需要外部传入标签，进行定制
  2. 封装：
  1. 使用<slot name=“名字”></slot>占位；
  2. 每个<slot>标签之间可以设置默认内容
  1. 使用：用标签<template #名字></template> or <template v-slot:名字>替换<slot>
vue
<!-- ChildComponent.vue -->
<template>
  <div class="child-container">
    <!-- 定义具名插槽，名称为 header -->
    <header>
      <slot name="header">默认标题</slot>   
    </header>

    <!-- 默认插槽 -->
    <main>
      <slot>默认内容</slot>
    </main>

    <!-- 定义具名插槽，名称为 footer -->
    <footer>
      <slot name="footer">默认页脚</slot>   
    </footer>
  </div>
</template>

<script>
export default {
  name: 'ChildComponent'
};
</script>

3. 作用域插槽 ​

  1. 定义：使用插槽<slot name="title" :age="18">传参数；
  2. 注意：传过去的是一个对象
vue
<!-- ChildComponent.vue -->
<template>
  <div class="child-container">
    <!-- 定义一个具名的作用域插槽 item，并传递 item 数据 -->
    <slot name="header" :data="itemData"></slot>
  </div>
</template>

<script>
export default {
  name: 'ChildComponent',
  data() {
    return {
      itemData: {
        name: 'Vue 插槽',
        description: '作用域插槽示例'
      }
    };
  }
};
</script>

      </div>

</template>

组件的样式冲突 ​

TIP

  1. 冲突：后面的组件样式会覆盖前面的样式
  2. 解决：在style标签上使用scoped属性，让样式只对当前组件生效
  3. 原理：
  1. 组件标签会增加一个data-v-hash属性；
  2. 样式会增加属性选择器data-v-hash
