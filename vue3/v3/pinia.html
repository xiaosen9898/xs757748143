Pinia ​

概念 ​

TIP

官方推荐使用，更好的状态管理工具

 

核心 ​

TIP

  1. state：数据，语法和作用对应组件内的data（是函数）
  2. getters：计算属性，语法和作用对应组件内的computed（是对象）
  3. actions：动作，语法和作用对应组件内的motheds（是对象）

 

配置使用 ​

TIP

  1. 安装：npm i pinia
  2. 配置使用：
js
 // stores/user.js
 import { defineStore } from 'pinia';
 
 export const useUserStore = defineStore('user', {
   state: () => ({
     name: 'Alice',
     age: 25
   }),
 
   getters: {
     isAdult: (state) => state.age >= 18
   },
 
   actions: {
     setName(newName) {
       this.name = newName;
     },
     incrementAge() {
       this.age++;
     }
   }
 });

 // stores/product.js
 import { defineStore } from 'pinia';
 
 export const useProductStore = defineStore('product', {
   state: () => ({
     products: [],
     productsDes: 'products',
     selectedProduct: null
   }),
 
   getters: {
     productCount: (state) => state.products.length
   },
 
   actions: {
     addProduct(product) {
       this.products.push(product);
     },
     selectProduct(productId) {
       this.selectedProduct = this.products.find(product => product.id === productId);
     }
   }
 });

核心-getters ​

TIP

  1. 定义：pinia中的getters和vuex中的基本是一样的，也带有缓存的功能
  2. 使用：
js
// src/store/counter.js
import { defineStore } from 'pinia';

export const useCounterStore = defineStore('counter', {
  state: () => ({
    count: 0
  }),
  
  // 定义 getters
  getters: {
    // 定义 doubleCount，返回 count 的两倍
    doubleCount: (state) => state.count * 2,
    
    // 也可以定义更复杂的 getter
    isEven: (state) => state.count % 2 === 0
  },
  
  // 定义 actions 用于更改 state
  actions: {
    increment() {
      this.count++;
    },
    decrement() {
      this.count--;
    }
  }
});

核心-actions ​

TIP

  1. 定义：在pinia中没有mutations，只有actions，同步和异步代码，都在actions中完成
  2. 使用：
js
// src/store/counter.js
import { defineStore } from 'pinia';

export const useCounterStore = defineStore('counter', {
  state: () => ({
    count: 0
  }),
  
  actions: {
    // 简单同步方法
    increment() {
      this.count++;
    },
    
    decrement() {
      this.count--;
    },
    
    // 带参数的方法
    incrementBy(value) {
      this.count += value;
    },
    
    // 异步操作
    async fetchCount() {
      try {
        // 模拟从 API 获取数据
        const response = await new Promise((resolve) => {
          setTimeout(() => resolve(10), 1000); // 假设 API 返回 10
        });
        this.count = response;
      } catch (error) {
        console.error("Failed to fetch count:", error);
      }
    }
  }
});

import { useCounterStore } from './store/modules/counter';

const counterStore = useCounterStore();

// 订阅状态变化
counterStore.$subscribe((mutation, state) => {
  console.log('State changed:', state);
  console.log('Mutation type:', mutation);
});

// 监听 Store 状态变化并执行副作用
// 例如，将状态存储到 localStorage
counterStore.$subscribe((mutation, state) => {
  localStorage.setItem('counter', JSON.stringify(state.count));
});
